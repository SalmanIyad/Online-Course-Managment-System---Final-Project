-- Student entity
CREATE TABLE Student (
  Student_ID NUMBER PRIMARY KEY,
  Name VARCHAR2(100),
  Email VARCHAR2(100),
  Password VARCHAR2(50)
);
DESCRIBE Student;
-- Table STUDENT created.


-- Course entity
CREATE TABLE Course (
  Course_ID NUMBER PRIMARY KEY,
  Title VARCHAR2(100),
  Description VARCHAR2(500),
  Duration NUMBER
);
DESCRIBE Course;
-- Table COURSE created.


-- Enrollment entity
CREATE TABLE Instructor (
  Instructor_ID NUMBER PRIMARY KEY,
  Name VARCHAR2(100),
  Email VARCHAR2(100),
  Specialization VARCHAR2(200)
);
DESCRIBE Instructor;
-- Table INSTRUCTOR created.


-- Enrollment entity
CREATE TABLE Enrollment (
    Enrollment_ID NUMBER PRIMARY KEY,
    Student_ID NUMBER,
    CONSTRAINT Student_ID_C FOREIGN KEY(Student_ID) REFERENCES Student(Student_ID),
    Course_ID NUMBER,
    CONSTRAINT Course_ID_C FOREIGN KEY (Course_ID) REFERENCES Course(Course_ID),
    Enrollment_Date DATE
);
DESCRIBE Enrollment;
-- Table ENROLLMENT created.

-- Create the lesson table
CREATE TABLE lesson (
    lesson_id INT PRIMARY KEY,
    course_id INT,
    lesson_title VARCHAR(255),
    lesson_description VARCHAR(255),
    lesson_date DATE,
    FOREIGN KEY (course_id) REFERENCES course(course_id)
);

-- Create the relationship between lesson and student (many-to-many)
CREATE TABLE student_lesson (
    student_id INT,
    lesson_id INT,
    PRIMARY KEY (student_id, lesson_id),
    FOREIGN KEY (student_id) REFERENCES student(student_id),
    FOREIGN KEY (lesson_id) REFERENCES lesson(lesson_id)
);

-- Teaches Relationship
CREATE TABLE Teaches (
    Instructor_ID NUMBER,
    CONSTRAINT Instructor_ID FOREIGN KEY (Instructor_ID) REFERENCES Instructor(Instructor_ID),
    Course_ID NUMBER,
    CONSTRAINT Course_ID FOREIGN KEY (Course_ID) REFERENCES Course(Course_ID),
    PRIMARY KEY (Instructor_ID, Course_ID)
);
DESCRIBE Teaches;
-- Table TEACHES created.


-- Enrolls Relationship
CREATE TABLE Enrolls (
    Student_ID NUMBER,
    CONSTRAINT Student_ID_C2 FOREIGN KEY (Student_ID) REFERENCES Student(Student_ID),
    Course_ID NUMBER,
    CONSTRAINT Course_ID_C2 FOREIGN KEY (Course_ID) REFERENCES Course(Course_ID),
    PRIMARY KEY (Student_ID, Course_ID)
);
DESCRIBE Enrolls;
-- Table ENROLLS created.
